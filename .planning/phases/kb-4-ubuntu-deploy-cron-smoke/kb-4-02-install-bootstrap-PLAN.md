    Write `kb/deploy/install.sh` (POSIX bash, `#!/usr/bin/env bash`, `set -euo pipefail`). Structure:

    ```bash
    #!/usr/bin/env bash
    # kb-api install/update bootstrap. Idempotent — safe to re-run.
    # Usage: sudo bash kb/deploy/install.sh
    #
    # Prerequisites checked (ALL must pass before any mutation):
    #   1. Running as root (sudo)
    #   2. Ubuntu 22.04+ (lsb_release -rs)
    #   3. SQLite >= 3.34 (FTS5 trigram tokenizer)
    #   4. Python >= 3.11
    #   5. Caddy installed
    #   6. /etc/systemd/system writable
    #   7. KB_DB_PATH source SQLite file exists (if env set; else default check)
    set -euo pipefail

    # ---- Configuration (operator may override via env before invoking) ----
    : "${KB_INSTALL_PREFIX:=/opt/OmniGraph-Vault}"
    : "${KB_USER:=kb}"
    : "${KB_DB_PATH:=/home/${KB_USER}/.hermes/data/kol_scan.db}"
    : "${KB_LOG_DIR:=/var/log/kb}"
    : "${KB_SYSTEMD_TARGET:=/etc/systemd/system/kb-api.service}"
    : "${KB_CADDY_INCLUDE_DOC:=/etc/caddy/conf.d/kb-api.snippet}"

    log()  { printf '[install] %s\n' "$*"; }
    fail() { printf '[install][FATAL] %s\n' "$*" >&2; exit 1; }

    # ---- Phase 1: Prerequisite checks (READ ONLY — must be before any mutation) ----

    log "Phase 1/3 — checking prerequisites"

    # 1.1 root
    [[ $EUID -eq 0 ]] || fail "must run as root (use sudo)"

    # 1.2 Ubuntu version
    if ! command -v lsb_release >/dev/null; then
      fail "lsb_release not found — install lsb-release package or run on Ubuntu"
    fi
    UBUNTU_REL="$(lsb_release -rs)"
    UBUNTU_MAJOR="${UBUNTU_REL%%.*}"
    [[ "${UBUNTU_MAJOR}" -ge 22 ]] || fail "Ubuntu 22.04+ required (found ${UBUNTU_REL})"

    # 1.3 SQLite >= 3.34
    if ! command -v sqlite3 >/dev/null; then
      fail "sqlite3 not found — apt install sqlite3"
    fi
    SQLITE_VER="$(sqlite3 --version | awk '{print $1}')"
    SQLITE_MAJOR="$(echo "${SQLITE_VER}" | cut -d. -f1)"
    SQLITE_MINOR="$(echo "${SQLITE_VER}" | cut -d. -f2)"
    if (( SQLITE_MAJOR < 3 )) || (( SQLITE_MAJOR == 3 && SQLITE_MINOR < 34 )); then
      fail "SQLite >= 3.34 required for FTS5 trigram (found ${SQLITE_VER})"
    fi

    # 1.4 Python >= 3.11
    if ! command -v python3 >/dev/null; then
      fail "python3 not found"
    fi
    PY_VER="$(python3 -c 'import sys; print(f"{sys.version_info[0]}.{sys.version_info[1]}")')"
    PY_MAJOR="${PY_VER%.*}"
    PY_MINOR="${PY_VER##*.}"
    if (( PY_MAJOR < 3 )) || (( PY_MAJOR == 3 && PY_MINOR < 11 )); then
      fail "Python >= 3.11 required (found ${PY_VER})"
    fi

    # 1.5 Caddy installed
    if ! command -v caddy >/dev/null; then
      fail "caddy not found — install Caddy first"
    fi

    # 1.6 systemd writable
    [[ -w /etc/systemd/system ]] || fail "/etc/systemd/system not writable (need root)"

    # 1.7 KB source files present
    [[ -f "${KB_INSTALL_PREFIX}/kb/deploy/kb-api.service" ]] || \
      fail "${KB_INSTALL_PREFIX}/kb/deploy/kb-api.service not found — clone repo first"
    [[ -f "${KB_INSTALL_PREFIX}/kb/deploy/Caddyfile.snippet" ]] || \
      fail "${KB_INSTALL_PREFIX}/kb/deploy/Caddyfile.snippet not found"
    # KB_DB_PATH may not exist on first install (operator scp's it after) — warn not fail
    if [[ ! -f "${KB_DB_PATH}" ]]; then
      log "WARN: KB_DB_PATH=${KB_DB_PATH} not found (you must populate before service starts)"
    fi

    log "Phase 1/3 OK — all prerequisites passed"

    # ---- Phase 2: Mutation (idempotent) ----

    log "Phase 2/3 — installing systemd unit + Caddy snippet"

    # 2.1 Create kb user if missing
    if ! id "${KB_USER}" >/dev/null 2>&1; then
      log "creating system user ${KB_USER}"
      useradd --system --home-dir "/home/${KB_USER}" --create-home --shell /usr/sbin/nologin "${KB_USER}"
    else
      log "user ${KB_USER} exists"
    fi

    # 2.2 Log dir
    mkdir -p "${KB_LOG_DIR}"
    chown "${KB_USER}:${KB_USER}" "${KB_LOG_DIR}"

    # 2.3 Install/update systemd unit (compare hash, skip cp if identical → idempotent)
    SRC_UNIT="${KB_INSTALL_PREFIX}/kb/deploy/kb-api.service"
    if [[ -f "${KB_SYSTEMD_TARGET}" ]] && cmp -s "${SRC_UNIT}" "${KB_SYSTEMD_TARGET}"; then
      log "systemd unit unchanged (idempotent — skipping cp)"
      UNIT_CHANGED=0
    else
      cp "${SRC_UNIT}" "${KB_SYSTEMD_TARGET}"
      log "systemd unit installed → ${KB_SYSTEMD_TARGET}"
      UNIT_CHANGED=1
    fi

    # 2.4 Install/update Caddy snippet (operator must include in main Caddyfile manually)
    mkdir -p "$(dirname "${KB_CADDY_INCLUDE_DOC}")"
    cp "${KB_INSTALL_PREFIX}/kb/deploy/Caddyfile.snippet" "${KB_CADDY_INCLUDE_DOC}"
    log "Caddy snippet copied → ${KB_CADDY_INCLUDE_DOC}"
    log "  ACTION REQUIRED: include this snippet in your main Caddyfile site block"
    log "  example: import ${KB_CADDY_INCLUDE_DOC}"

    # 2.5 systemd reload + enable
    systemctl daemon-reload
    if (( UNIT_CHANGED == 1 )); then
      log "unit changed — restart triggered after enable"
    fi
    systemctl enable --now kb-api.service

    # 2.6 Caddy reload (best-effort — operator-managed snippet inclusion may not be active)
    if systemctl is-active --quiet caddy; then
      log "reloading caddy"
      systemctl reload caddy || log "WARN: caddy reload failed (verify snippet inclusion + caddy validate)"
    else
      log "WARN: caddy service not active — skipping reload (operator must start caddy)"
    fi

    log "Phase 2/3 OK"

    # ---- Phase 3: Post-install verification ----

    log "Phase 3/3 — verification"

    sleep 2  # give uvicorn time to bind

    if systemctl is-active --quiet kb-api.service; then
      log "✓ kb-api.service: active"
    else
      log "✗ kb-api.service NOT active — check journalctl -u kb-api"
      exit 2
    fi

    # Health probe (fail soft — caller may not have curl)
    if command -v curl >/dev/null; then
      if curl -fsS http://127.0.0.1:8766/health >/dev/null 2>&1; then
        log "✓ /health returned 2xx"
      else
        log "WARN: /health probe failed — check journalctl -u kb-api"
      fi
    fi

    log "Phase 3/3 OK"
    log ""
    log "Install complete. Next steps:"
    log "  - Include ${KB_CADDY_INCLUDE_DOC} in your main Caddyfile + reload"
    log "  - Set up daily cron: see kb/scripts/daily_rebuild.sh (kb-4-04)"
    log "  - Run smoke verification: see .planning/phases/kb-4-ubuntu-deploy-cron-smoke/"
    ```
