Step 2.3 — Write `.scratch/repro44/run_repro44.py` (target ≤120 LoC). Pattern adapted from `.scratch/260606-bd-cache-probe-vertex.py`:

```python
"""260609-eg1 — reproduce ISSUES #44 entity-extract 0-entity silent failure on local Corp Vertex Gemini.

Usage:
    python .scratch/repro44/run_repro44.py <hash1> <hash2> <hash3>

Reads body for each hash from .scratch/repro44/{hash}.json (article.body field).
For each doc:
  1. Wipe .dev-runtime/repro44/{hash}/lightrag_storage (clean slate per doc — NO cross-doc contamination)
  2. Build LightRAG with working_dir = that storage, llm = Vertex Gemini, embedding = lib.embedding_func
  3. await rag.ainsert(body, ids=[hash])
  4. Walk graph_chunk_entity_relation.graphml, count nodes whose source_id contains the hash
  5. Read kv_store_doc_status.json[hash] for status + chunks_count
  6. Emit JSON: {hash, body_len, image_count, wall_s, exception, status, entity_count, chunks_count}
  7. Write per-doc log to {hash}.log + per-doc result JSON to {hash}-result.json

NO production source mutation. NO Aliyun touch. NO cross-doc shared state.
"""
import asyncio, json, os, shutil, sys, time, traceback, xml.etree.ElementTree as ET
from collections import Counter
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent.parent
sys.path.insert(0, str(REPO))

# Force Vertex Gemini provider before any lib import that reads the env
os.environ.setdefault("OMNIGRAPH_LLM_PROVIDER", "vertex_gemini")

from lightrag import LightRAG
from lightrag.kg.shared_storage import initialize_pipeline_status
from lib import embedding_func
from lib.llm_complete import get_llm_func

OUT = Path(".scratch/repro44").resolve()
RUNTIME = Path(os.environ.get("OMNIGRAPH_BASE_DIR", ".dev-runtime/repro44")).resolve()


def count_entities_for(storage: Path, doc_hash: str) -> tuple[int, str]:
    """Return (entity_node_count, graphml_status). status ∈ {'ok','missing','corrupt:<msg>'}."""
    g = storage / "graph_chunk_entity_relation.graphml"
    if not g.exists():
        return 0, "missing"
    try:
        ns = {"g": "http://graphml.graphdrawing.org/xmlns"}
        tree = ET.parse(str(g))
        root = tree.getroot()
        # discover source_id attr key
        src_key = None
        for k in root.findall("g:key", ns):
            if k.get("attr.name") == "source_id":
                src_key = k.get("id"); break
        if not src_key:
            return 0, "no-source_id-key"
        n = 0
        for node in root.findall("g:graph/g:node", ns):
            for data in node.findall("g:data", ns):
                if data.get("key") == src_key and data.text and doc_hash in data.text:
                    n += 1; break
        return n, "ok"
    except Exception as e:
        return 0, f"corrupt:{e}"


async def ingest_one(doc_hash: str, body: str, storage: Path) -> dict:
    if storage.exists():
        shutil.rmtree(storage)
    storage.mkdir(parents=True, exist_ok=True)

    rag = LightRAG(
        working_dir=str(storage),
        llm_model_func=get_llm_func(),
        embedding_func=embedding_func,
        default_embedding_timeout=int(os.environ.get("LIGHTRAG_EMBEDDING_TIMEOUT", "180")),
    )
    await rag.initialize_storages()
    await initialize_pipeline_status()

    res = {"hash": doc_hash, "body_len": len(body)}
    t0 = time.monotonic()
    try:
        await rag.ainsert(body, ids=[doc_hash])
        res["exception"] = None
    except Exception as e:
        res["exception"] = f"{type(e).__name__}: {e}\n{traceback.format_exc()}"
    res["wall_s"] = round(time.monotonic() - t0, 2)

    # Read doc_status
    ds_path = storage / "kv_store_doc_status.json"
    try:
        ds = json.loads(ds_path.read_text(encoding="utf-8"))
        entry = ds.get(doc_hash) or {}
        res["status"] = entry.get("status")
        res["chunks_count"] = entry.get("chunks_count")
    except Exception as e:
        res["status"] = f"read-error:{e}"
        res["chunks_count"] = None

    n_ent, gstatus = count_entities_for(storage, doc_hash)
    res["entity_count"] = n_ent
    res["graphml_status"] = gstatus

    await rag.finalize_storages()
    return res


async def main(hashes: list[str]):
    OUT.mkdir(parents=True, exist_ok=True)
    RUNTIME.mkdir(parents=True, exist_ok=True)

    summary = {"hashes": hashes, "results": []}
    for h in hashes:
        src = OUT / f"{h}.json"
        data = json.loads(src.read_text(encoding="utf-8"))
        body = (data.get("article") or {}).get("body") or ""
        if len(body) < 500:
            print(f"WARN: {h} body too short ({len(body)} chars), skipping (would be filtered prod-side)")
            summary["results"].append({"hash": h, "skipped": "body<500"})
            continue
        storage = RUNTIME / h / "lightrag_storage"
        print(f">>> ingest {h} body_len={len(body)} storage={storage}")
        result = await ingest_one(h, body, storage)
        (OUT / f"{h}-result.json").write_text(json.dumps(result, indent=2, default=str), encoding="utf-8")
        summary["results"].append(result)
        print(f"<<< {h} entity_count={result['entity_count']} status={result.get('status')} wall_s={result.get('wall_s')}")

    print("\n=== SUMMARY ===")
    print(json.dumps(summary, indent=2, default=str))


if __name__ == "__main__":
    if len(sys.argv) < 4:
        sys.exit("usage: run_repro44.py <hash1> <hash2> <hash3>")
    asyncio.run(main(sys.argv[1:4]))
```

The script uses `.dev-runtime/repro44/{hash}/lightrag_storage` as **per-doc isolated working dir** so docs cannot contaminate each other's graphml.
