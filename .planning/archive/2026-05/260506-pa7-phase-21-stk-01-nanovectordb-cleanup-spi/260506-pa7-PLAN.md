```
"""STK-01 spike: probe whether rag.adelete_by_doc_id cleans all 4 storage layers
when a doc is in FAILED status. Read-mostly. Snapshot taken in Step A wrapper.

Run via: venv\\Scripts\\python .dev-runtime\\run_local.py .planning/quick/.../spike_cleanup_probe.py
"""
import asyncio, json, os, time, shutil, sys
from pathlib import Path
import xml.etree.ElementTree as ET

ROOT = Path(__file__).resolve().parents[3]  # repo root
STORAGE = ROOT / ".dev-runtime" / "lightrag_storage"
PROBE_TAG = f"STK-01-PROBE-{int(time.time())}"
PROBE_TEXT = f"This is a {PROBE_TAG} diagnostic doc. It has no real content. Generated for Phase 21 STK-01 cleanup-completeness probe."
# ids= forces a known doc_id; LightRAG normalizes the prefix internally
PROBE_DOC_ID = f"stk01-probe-{int(time.time())}"

def _atomic_write_json(path: Path, data: dict) -> None:
    """Per CLAUDE.md atomic-write convention: write .tmp then os.replace."""
    tmp = path.with_suffix(path.suffix + ".tmp")
    tmp.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
    os.replace(tmp, path)

def _read_json(path: Path) -> dict:
    if not path.exists(): return {}
    return json.loads(path.read_text(encoding="utf-8"))

def _probe_doc_status(doc_id: str) -> dict:
    """Layer 1 probe: kv_store_doc_status.json"""
    data = _read_json(STORAGE / "kv_store_doc_status.json")
    return {"present": doc_id in data, "raw": data.get(doc_id)}

def _probe_full_docs(doc_id: str) -> dict:
    """Layer 2 probe: kv_store_full_docs.json"""
    data = _read_json(STORAGE / "kv_store_full_docs.json")
    return {"present": doc_id in data, "size": len(str(data.get(doc_id, "")))}

def _probe_vdb_entities(doc_id: str, tag: str) -> dict:
    """Layer 3 probe: vdb_entities.json — search for entity rows whose source_id
    contains doc_id, AND for any row whose content/description contains the
    PROBE_TAG (catches entities extracted from probe text)."""
    data = _read_json(STORAGE / "vdb_entities.json")
    rows = data.get("data", []) if isinstance(data, dict) else data
    by_source = [r for r in rows if isinstance(r, dict) and doc_id in str(r.get("source_id", ""))]
    by_tag    = [r for r in rows if isinstance(r, dict) and tag in json.dumps(r, ensure_ascii=False)]
    return {"by_source_count": len(by_source), "by_tag_count": len(by_tag),
            "sample_by_source": by_source[:2], "sample_by_tag": by_tag[:2]}

def _probe_graphml(doc_id: str, tag: str) -> dict:
    """Layer 4 probe: graph_chunk_entity_relation.graphml — search for nodes/edges
    whose source_id attribute references the doc_id, OR whose description text
    contains the PROBE_TAG."""
    path = STORAGE / "graph_chunk_entity_relation.graphml"
    if not path.exists(): return {"file_present": False}
    tree = ET.parse(path)
    root = tree.getroot()
    ns = {"g": "http://graphml.graphdrawing.org/xmlns"}
    matched_nodes = []
    matched_edges = []
    for node in root.iter("{http://graphml.graphdrawing.org/xmlns}node"):
        text = ET.tostring(node, encoding="unicode")
        if doc_id in text or tag in text:
            matched_nodes.append(node.get("id"))
    for edge in root.iter("{http://graphml.graphdrawing.org/xmlns}edge"):
        text = ET.tostring(edge, encoding="unicode")
        if doc_id in text or tag in text:
            matched_edges.append((edge.get("source"), edge.get("target")))
    return {"file_present": True, "matched_node_ids": matched_nodes[:5],
            "matched_node_count": len(matched_nodes),
            "matched_edge_count": len(matched_edges)}

async def main() -> dict:
    # Defer LightRAG import until inside main so import-time env mutations from
    # .dev-runtime/.env via run_local.py have already landed.
    sys.path.insert(0, str(ROOT))
    from lib.lightrag_factory import get_rag  # adjust import to whatever module exposes get_rag in this repo

    # 1. Insert probe doc with known id
    rag = await get_rag(flush=False)
    print(f"[1/6] Inserting probe doc id={PROBE_DOC_ID} tag={PROBE_TAG}")
    await rag.ainsert(PROBE_TEXT, ids=[PROBE_DOC_ID])

    # 2. Confirm it's in storage
    pre_status = _probe_doc_status(PROBE_DOC_ID)
    pre_full   = _probe_full_docs(PROBE_DOC_ID)
    print(f"[2/6] Post-insert: doc_status.present={pre_status['present']} full_docs.present={pre_full['present']}")
    assert pre_status["present"], "probe doc not in doc_status after ainsert — abort"

    # 3. Force FAILED status (atomic write)
    print(f"[3/6] Forcing status=failed via direct doc_status edit (atomic .tmp + replace)")
    status_path = STORAGE / "kv_store_doc_status.json"
    data = _read_json(status_path)
    if isinstance(data.get(PROBE_DOC_ID), dict):
        data[PROBE_DOC_ID]["status"] = "failed"
    else:
        # Some versions store the value differently — log it and abort cleanly
        print(f"  WARN: doc_status entry shape unexpected: {data.get(PROBE_DOC_ID)!r}")
    _atomic_write_json(status_path, data)

    # 4. Call the API under test
    print(f"[4/6] Calling rag.adelete_by_doc_id({PROBE_DOC_ID})")
    delete_exception = None
    try:
        await rag.adelete_by_doc_id(PROBE_DOC_ID)
    except Exception as exc:
        delete_exception = repr(exc)
        print(f"  EXCEPTION from adelete_by_doc_id: {delete_exception}")

    # 5. Probe all 4 layers for residue
    print(f"[5/6] Probing 4 storage layers for residue")
    findings = {
        "probe_doc_id": PROBE_DOC_ID,
        "probe_tag": PROBE_TAG,
        "delete_exception": delete_exception,
        "layer_1_doc_status":   _probe_doc_status(PROBE_DOC_ID),
        "layer_2_full_docs":    _probe_full_docs(PROBE_DOC_ID),
        "layer_3_vdb_entities": _probe_vdb_entities(PROBE_DOC_ID, PROBE_TAG),
        "layer_4_graphml":      _probe_graphml(PROBE_DOC_ID, PROBE_TAG),
    }
    # Also probe the other vdb files + entity_chunks because they SHOULD be checked
    # for completeness even though user spec didn't list them — finding scope
    # creep here is cheap.
    findings["bonus_vdb_chunks"]      = _probe_vdb_entities(PROBE_DOC_ID, PROBE_TAG)  # reuse helper, just point at chunks file
    # Actually: write a tiny inline probe for vdb_chunks + kv_store_text_chunks so
    # findings doc captures the full picture.
    for extra in ("vdb_chunks.json", "vdb_relationships.json", "kv_store_text_chunks.json",
                  "kv_store_entity_chunks.json", "kv_store_relation_chunks.json"):
        raw = _read_json(STORAGE / extra)
        text = json.dumps(raw, ensure_ascii=False)
        findings[f"bonus_{extra}"] = {
            "doc_id_hits": text.count(PROBE_DOC_ID),
            "tag_hits": text.count(PROBE_TAG),
        }

    # 6. Verdict
    layers_with_residue = []
    if findings["layer_1_doc_status"]["present"]: layers_with_residue.append("kv_store_doc_status.json")
    if findings["layer_2_full_docs"]["present"]:  layers_with_residue.append("kv_store_full_docs.json")
    if findings["layer_3_vdb_entities"]["by_source_count"] or findings["layer_3_vdb_entities"]["by_tag_count"]:
        layers_with_residue.append("vdb_entities.json")
    if findings["layer_4_graphml"].get("matched_node_count", 0) or findings["layer_4_graphml"].get("matched_edge_count", 0):
        layers_with_residue.append("graph_chunk_entity_relation.graphml")
    for k, v in findings.items():
        if k.startswith("bonus_") and isinstance(v, dict) and (v.get("doc_id_hits", 0) or v.get("tag_hits", 0)):
            layers_with_residue.append(k.replace("bonus_", ""))

    if not layers_with_residue:
        verdict = "cleanup 完整 — adelete_by_doc_id removes residue from all probed layers"
    else:
        verdict = f"cleanup 残留 in {len(layers_with_residue)} 层: {', '.join(layers_with_residue)}"
    findings["verdict"] = verdict

    # Sanity: fixture docs still present?
    full = _read_json(STORAGE / "kv_store_full_docs.json")
    findings["fixture_doc_count_after_spike"] = len(full)

    print(f"[6/6] VERDICT: {verdict}")
    return findings

if __name__ == "__main__":
    out = asyncio.run(main())
    print("\n\n=== FINDINGS JSON ===")
    print(json.dumps(out, ensure_ascii=False, indent=2))
```
